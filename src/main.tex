	\section{Preliminaries}

	\subsection{Notations}

	\subsubsection{Constructing tuples}

	Let \(c_1, c_2, \cdots, c_k\) be constants.

	\(c_1, c_2, \cdots, c_k\) may be arranged into a \(k\)-tuple as follow:
	\[
		(c_1, c_2, \cdots, c_k)
	\]

	\subsubsection{Indexing tuples}

	Let \(C\) be a \(k\)-tuple.

	\(C(i)\) is the \(i\)-th element in tuple \(C\).

	\subsubsection{Slicing tuples}

	Let \(U\) be a set where \(|U| = k\).

	For \(U\) to have correspondence with \(C\) means there exists a bijective function \(\mathtt{idx} : U \to \{1, \cdots, k\}\).

	If \(u \in U\), then \(C(u) = C(\mathtt{idx}(u))\).

	If \(\{u_1, \cdots, u_r\} \subseteq U\), then, preserving the element order of \(C\),
	\[
		C(\{u_1,\cdots, u_r\}) = (C(u_1), \cdots, C(u_r))
	\]

	\subsubsection{Masking sets}

	Let \(B\) be a \(k\)-Boolean tuple.

	If \(U\) has a correspondence with \(B\), then
	\[
		U(B) = \{ u \in U : B(u) = \mathtt{True}\}
	\]

	\subsubsection{Operations on tuples}

	\[
		\neg B = (\neg B(1), \cdots, \neg B(k))
	\]

	Let \(B_1\) and \(B_2\) be \(k\)-Boolean tuples.

	\[
		B_1 \vee B_2 = (B_1(1) \vee B_2(1), \cdots, B_1(k) \vee B_2(k))
	\]

	Let \(C_1\) and \(C_2\) be \(k\)-tuples.

	\[
		( C_1 = C_2 ) = (C_1(1) = C_2(1), \cdots, C_1(k) = C_2(k))
	\]

	\subsubsection{Constructing matrices}

	Let \(a_1, a_2, \cdots, a_n\) be \(m\)-tuples.

	\(a_1, a_2, \cdots, a_n\) may be arranged into the rows of an \(n \times m\) matrix as follow:
	\[
		\left[
		\begin{array}{c}
			a_1 \\
			a_2 \\
			\vdots \\
			a_n
		\end{array}
		\right]
	\]

	\(a_1, a_2, \cdots, a_n\) may be arranged into the columns of an \(m \times n\) matrix as follow:
	\[
		\left[ a_1\enspace a_2\enspace \cdots\enspace a_n \right]
	\]

	Let \(A_1\) be an \(n_1 \times m\) matrix, \(A_2\) be an \(n_2 \times m\) matrix, \(\cdots\), and \(A_r\) be an \(n_r \times m\) matrix.

	\(A_1\), \(A_2\), \(\cdots\), \(A_r\) may be stacked vertically into an \((n_1 + n_2 + \cdots + n_r) \times m\) matrix as follow:
	\[
		\left[
		\begin{array}{c}
			A_1 \\
			A_2 \\
			\vdots \\
			A_r
		\end{array}
		\right]
	\]

	Let \(A'_1\) be an \(n \times m_1\) matrix, \(A'_2\) be an \(n \times m_2\) matrix, \(\cdots\), and \(A'_r\) be an \(n \times m_r\) matrix.

	\(A'_1\), \(A'_2\), \(\cdots\), \(A'_r\) may be stacked horizontally into an \(n \times (m_1 + m_2 + \cdots + m_r)\) matrix as follow:
	\[
		\left[ A'_1\enspace A'_2\enspace \cdots\enspace A'_r \right]
	\]

	\subsubsection{Indexing matrices}

	Let \(A\) be an \(n \times m\) matrix.

	\(A(i, j)\) is the element at the \(i\)-th row and \(j\)-th column in matrix \(A\).

	\[ A(i, \star) = (A(i, 1), \cdots, A(i, m)) \]

	\[ A(\star, j) = (A(1, j), \cdots, A(n, j)) \]

	\subsubsection{Slicing matrices}

	Let \(S\) be a set where \(|S| = n\).

	For \(S\) to have row correspondence with \(A\) means there exists a bijective function \(\mathtt{idx} : S \to \{1, \cdots, n\}\).

	If \(s \in S\), then \(A(s, \star) = A(\mathtt{idx}(s), \star)\).

	If \(\{s_1, \cdots, s_r\} \subseteq S\), then, preserving the row order of \(A\),
	\[
		A(\{s_1, \cdots, s_r\}, \star) = \left[
		\begin{array}{c}
			A(s_1, \star) \\
			\vdots \\
			A(s_r, \star)
		\end{array}
		\right]
	\]

	Let \(T\) be a set where \(|T| = m\).

	For \(T\) to have column correspondence with \(A\) means there exists a bijective function \(\mathtt{idx} : T \to \{1, \cdots, m\}\).

	If \(t \in T\), then \(A(\star, t) = A(\star, \mathtt{idx}(t))\).

	If \(\{t_1, \cdots, t_r\} \subseteq T\), then, preserving the column order of \(A\),
	\[
		A(\star, \{t_1, \cdots, t_r\}) = \left[ A(\star, t_1)\enspace \cdots\enspace A(\star, t_r) \right]
	\]

	\subsection{Graphs}

	\subsubsection{Distance}

	Let \(G\) be a graph.

	Let \(V\) be the set of vertices in \(G\).

	Let \(E\) be the set of edges in \(G\).

	For \(u, v \in V\), the distance from \(u\) to \(v\) is denoted by \(d(u, v)\).

	For \(u, v, w \in V\) and \(e \in E\), if \(e\) is an edge between \(v\) and \(w\), then the distance from \(u\) to \(e\) is
	\[
		d(u, e) = \mathrm{min}(d(u, v), d(u, w))
	\]

	\subsubsection{Representation}

	For \(v \in V\) and \(W = \{w_1, w_2, \cdots, w_k\} \subseteq V\), the representation of \(v\) with respect to \(W\) is
	\[
		r(v|W) = (d(w_1, v), d(w_2, v), \cdots, d(w_k, v))
	\]

	For \(e \in E\) and \(W = \{w_1, w_2, \cdots, w_k\} \subseteq V\), the representation of \(e\) with respect to \(W\) is
	\[
		r(e|W) = (d(w_1, e), d(w_2, e), \cdots, d(w_k, e))
	\]

	\subsubsection{Resolving set}

	For \(W\) to be a resolving set means that
	\[
		\forall u, v \in V, u \neq v \implies r(u|W) \neq r(v|W)
	\]

	For \(W\) to be an edge resolving set means that
	\[
		\forall e, f \in E, e \neq f \implies r(e|W) \neq r(f|W)
	\]

	\subsubsection{Non-resolving set}

	If \(W\) is not a resolving set, then \(W\) is a non-resolving set.

	\(\mathcal{P}(V)\) is partitioned into the set of resolving sets and the set of non-resolving sets.

	If \(W\) is not an edge resolving set, then \(W\) is an edge non-resolving set.

	\(\mathcal{P}(V)\) is partitioned into the set of edge resolving sets and the set of edge non-resolving sets.

	\subsubsection{Monotonicity}

	If \(W\) is a resolving set, then a superset of \(W\) is a resolving set.

	If \(W\) is a non-resolving set, then a subset of \(W\) is a non-resolving set.

	If \(W\) is an edge resolving set, then a superset of \(W\) is an edge resolving set.

	If \(W\) is an edge non-resolving set, then a subset of \(W\) is an edge non-resolving set.

	\subsubsection{Metric dimension}

	The metric dimension of a graph is the minimum cardinality of resolving sets of the graph.

	The edge metric dimension of a graph is the minimum cardinality of edge resolving sets of the graph.

	\section{Distance Matrix}

	If \(V = \{v_1, v_2, \cdots, v_n\}\), then the distance matrix of \(G\) has dimension \(n \times n\), and is as follow:
	\[
		D^{(V)} = \left[
		\begin{array}{cccc}
			d(v_1, v_1) & d(v_1, v_2) & \cdots & d(v_1, v_n) \\
			d(v_2, v_1) & d(v_2, v_2) & \cdots & d(v_2, v_n) \\
			\vdots & \vdots & \ddots & \vdots \\
			d(v_n, v_1) & d(v_n, v_2) & \cdots & d(v_n, v_n)
		\end{array}
		\right]
	\]

	\(V\) has row correspondence and column correspondence with \(D^{(V)}\).

	Thus, for \(v \in V\) and \(W \subseteq V\), the representation of \(v\) with respect to \(W\) is
	\[
		r(v|W) = D^{(V)}(W, \star)(\star, v)
	\]

	If \(E = \{e_1, e_2, \cdots, e_m\}\), then the edge distance matrix of \(G\) has dimension \(n \times m\), and is as follow:
	\[
		D^{(E)} = \left[
		\begin{array}{cccc}
			d(v_1, e_1) & d(v_1, e_2) & \cdots & d(v_1, e_m) \\
			d(v_2, e_1) & d(v_2, e_2) & \cdots & d(v_2, e_m) \\
			\vdots & \vdots & \ddots & \vdots \\
			d(v_n, e_1) & d(v_n, e_2) & \cdots & d(v_n, e_m)
		\end{array}
		\right]
	\]

	\(V\) has row correspondence and \(E\) has column correspondence with \(D^{(E)}\).

	Thus, for \(e \in E\) and \(W \subseteq V\), the representation of \(e\) with respect to \(W\) is
	\[
		r(e|W) = D^{(E)}(W, \star)(\star, e)
	\]
